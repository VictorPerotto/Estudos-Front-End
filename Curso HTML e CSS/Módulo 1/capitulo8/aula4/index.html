Códigos / Pré-Formatações

Podemos introduzir uma linha de código como GameObject.GetComponent() marcando com a tag Code

        
public class GetComponentExample : MonoBehaviour
{
    public GameObject otherGameObject;

    void Start()
    {
        HingeJoint hinge = otherGameObject.GetComponent();

        if (hinge != null)
        {
            hinge.useSpring = false;
        }
    }
}
